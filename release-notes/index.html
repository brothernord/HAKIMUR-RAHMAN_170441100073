    Implementasi K-NN dengan Python¶

    Berikut kode program beserta penjelasannya

    #memuat/membuat library
from sklearn.datasets import load_iris
from sklearn.model_selection import train_test_split
import matplotlib.pyplot as plt
import pandas as pd
from sklearn.neighbors import KNeighborsClassifier
import numpy as np
from pandas.plotting import scatter_matrix


#dibawah ini memuat/memamnggil data dengan load_iris dari sklearn
iris_dataset = load_iris()

# menampilkan informasi tentang data iris
print("Keys of iris_dataset \n{}".format(iris_dataset.keys()))
print(iris_dataset['DESCR'][:193] + "\n...")
print(" Nama Kelas: {}".format(iris_dataset['target_names']))
print("Nama Fitur: {}".format(iris_dataset['feature_names']))
print("Type of data: {}".format(type(iris_dataset['data'])))
print("Shape of data: {}".format(iris_dataset['data'].shape))

# menampilkan 10 data iris untuk digunakan sebagai data training
print("Data Training: \n{}".format(iris_dataset['data'][:10]))

# Type of target
print("Type of target: {}".format(type(iris_dataset['target'])))
print("Shape of target: {}".format(iris_dataset['target'].shape))

# untuk hasil tampilannya nanti : 0 = Setosa, 1 = Versicolor, 2 = Virginica
print("Target: \n{}".format(iris_dataset['target']))

# Training and Testing Data
X_train, X_test, y_train, y_test = train_test_split(iris_dataset['data'],
                                                    iris_dataset['target'],
                                                    random_state=0)
print("X_train shape: {}".format(X_train.shape))
print("y_train shape: {}".format(y_train.shape))
print("X_test shape: {}".format(X_test.shape))
print("y_test shape: {}".format(y_test.shape))

# membuat dataframe dari data X_train
#memberi label pada kolom menggunakan string di iris_dataset.feature_names
iris_dataframe = pd.DataFrame(X_train, columns=iris_dataset.feature_names)

# membuat matriks pencar dari bingkai data, mewarnai y_train
scatter_matrix(iris_dataframe, c=y_train, figsize=(15, 15), marker='o',
                      hist_kwds={'bins': 20}, s=60, alpha=.8)
plt.show()

# membuat model menggunakan KNN
knn = KNeighborsClassifier(n_neighbors=1)
knn.fit(X_train, y_train)

X_new = np.array([[5, 2.9, 1, 0.2]])
print("X_new.shape: {}".format(X_new.shape))

prediction = knn.predict(X_new)
print("Prediksi: {}".format(prediction))
print("Nama target yang diprediksi: {}".format(iris_dataset['target_names'][prediction]))

# Evaluating model
y_pred = knn.predict(X_test)
print("kumpulan tes prediksi: \n {}".format(y_pred))
print("hasil set tes: {:.2f}".format(np.mean(y_pred == y_test)))
print("hasil set tes: {:.2f}".format(knn.score(X_test, y_test)))
